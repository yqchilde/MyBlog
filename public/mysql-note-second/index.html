DELIMITER //
CREATE PROCEDURE p_stu1(IN
	student_name char(10),IN
	course_name char(20))
BEGIN
	SELECT Sname, Cname, Grade
	FROM xs s JOIN xk ON s.sno = xk.sno
	INNER JOIN kc c ON c.cno = xk.cno
	WHERE sname = student_name
	AND cname = course_name;
END //
DELIMITER;

调用存储过程
